<div className="nano-quickstart">
  <h2>Quick Start</h2>

```js
import { create } from 'nano-css';
import { addon as addonRule } from 'nano-css/addon/rule';

const nano = create();
addonRule(nano);

const className = nano.rule({
  color: 'tomato',
  ':hover': {
    color: 'blue',
  },
});
```
